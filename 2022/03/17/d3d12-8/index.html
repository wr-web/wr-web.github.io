// 如果没有把所有可能包含进去会报错，不是很理解成因
float kdtrans(float kd)
{
    if(kd <= 0.0f){
        return 0.4f;
    }else if(kd <= 0.5f){
        return 0.6f;
    }else if(kd <= 1.0f){
        return 1.0f;
    }else{
        return 1.0f;
    }
}

float kstrans(float ks)
{
    if(ks <= 0.0f){
        return 0.0f;
    }else if(ks <= 0.1f){
        return 0.0f;
    }else if(ks <= 0.8f){
        return 0.5f;
    }else if(ks <= 1.0f){
        return 0.8f;
    }else{
        return 0.8f;
    }
}
//修改直接光模型
float ndotl = kdtrans(max(dot(lightVec, normal), 0.0f));
//修改BlinnPhong模型
float roughnessFactor = (m + 8.0f)*kstrans(pow(max(dot(halfVec, normal), 0.0f), m)) / 8.0f;
